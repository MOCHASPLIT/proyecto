//Ejecutar sql

CREATE TABLE sortix83_guerrero.users (
  id INT PRIMARY KEY AUTO_INCREMENT,
  username VARCHAR(50),
  password VARCHAR(100)
);

CREATE TABLE sortix83_guerrero.tasks (
  id INT PRIMARY KEY AUTO_INCREMENT,
  title VARCHAR(50),
  description VARCHAR(255),
  status BOOLEAN,
  user_id INT,
  FOREIGN KEY (user_id) REFERENCES users(id)
);
